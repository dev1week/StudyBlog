단순히 값을 찾기


int binarySearch(int[] arr, int target){  
  
    int l = 0;  
    int h = arr.length;  
  
    while(h>l){  
        int mid = (l+h)/2;  
        if(arr[mid]==target){  
            return mid;  
        }else if(arr[mid]>target){  
            h = mid;  
        }else{  
            l = mid+1;  
        }  
    }  
    return -1;  
}

가장 오른쪽 인덱스 구하기

int upperIdx(int target, int[] arr){
	int l = 0;
	int h = arr.length;
	while(h>l){
		int mid = (l+h)/2;
		if(arr[mid]>target) h = mid;
		else l = mid+1;
	}
	
	return l;

}

가장 왼쪽 인덱스 구하기

int lowerIdx(int target, int[] arr){
	int l = 0;
	int h = arr.length;
	while(h>l){
		int mid = (l+h)/2;
		if(arr[mid]>=target) h = mid;
		else l = mid+1;
	}
	
	return l;

}
